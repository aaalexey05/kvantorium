Урок 11: Объектно-ориентированное программирование (ООП)

Программируем с помощью объектов и классов

Что такое ООП?

ООП — это подход к программированию, в котором программа состоит из объектов, взаимодействующих друг с другом.

Класс

Чертеж или шаблон для создания объектов

Объект

Экземпляр класса с конкретными данными

Инкапсуляция

Сокрытие внутренней реализации

Наследование

Создание новых классов на основе существующих

Полиморфизм

Разное поведение у объектов с одинаковым интерфейсом

Классы и объекты

Класс описывает тип объектов, а объект — конкретный экземпляр этого типа.

# Простой класс
class Студент:
    # Конструктор (инициализатор)
    def __init__(self, имя, возраст, группа):
        # Атрибуты (свойства) объекта
        self.имя = имя
        self.возраст = возраст
        self.группа = группа
        self.оценки = []  # Начальное значение
    
    # Методы (функции объекта)
    def представитьсь(self):
        print(f"Привет! Меня зовут {self.имя}, мне {self.возраст} лет.")
        print(f"Я учусь в группе {self.группа}.")
    
    def добавить_оценку(self, оценка):
        self.оценки.append(оценка)
        print(f"Добавлена оценка: {оценка}")
    
    def средний_балл(self):
        if not self.оценки:
            return 0
        return sum(self.оценки) / len(self.оценки)

# Создание объектов (экземпляров класса)
студент1 = Студент("Анна", 12, "7А")
студент2 = Студент("Иван", 13, "7Б")
студент3 = Студент("Мария", 12, "7А")

# Использование объектов
студент1.представитьсь()  # Привет! Меня зовут Анна...
студент2.представитьсь()  # Привет! Меня зовут Иван...

# Работа с атрибутами
print(f"Имя студента1: {студент1.имя}")
print(f"Группа студента2: {студент2.группа}")

# Изменение атрибутов
студент1.возраст = 13
print(f"Новый возраст Анны: {студент1.возраст}")

# Использование методов
студент1.добавить_оценку(5)
студент1.добавить_оценку(4)
студент1.добавить_оценку(5)
print(f"Средний балл Анны: {студент1.средний_балл():.2f}")  # 4.67

Магические методы (dunder methods)

Специальные методы с двойным подчеркиванием, которые вызываются автоматически.

class Книга:
    def __init__(self, название, автор, год):
        self.название = название
        self.автор = автор
        self.год = год
        self.прочитано = False
    
    # __str__ - для str() и print()
    def __str__(self):
        статус = "прочитана" if self.прочитано else "не прочитана"
        return f'"{self.название}" - {self.автор} ({self.год}) [{статус}]'
    
    # __repr__ - для отладки и repr()
    def __repr__(self):
        return f"Книга('{self.название}', '{self.автор}', {self.год})"
    
    # __len__ - для len()
    def __len__(self):
        # Предположим, что у нас есть страницы
        return 200  # Просто пример
    
    # __eq__ - для сравнения ==
    def __eq__(self, другая_книга):
        if not isinstance(другая_книга, Книга):
            return False
        return (self.название == другая_книга.название and 
                self.автор == другая_книга.автор)
    
    # __lt__ - для сравнения <
    def __lt__(self, другая_книга):
        return self.год < другая_книга.год
    
    # __call__ - чтобы объект можно было вызывать как функцию
    def __call__(self):
        print(f"Читаем книгу: {self.название}")
        self.прочитано = True

# Использование
книга1 = Книга("Гарри Поттер", "Дж. К. Роулинг", 1997)
книга2 = Книга("Властелин колец", "Дж. Р. Р. Толкин", 1954)

print(str(книга1))      # "Гарри Поттер" - Дж. К. Роулинг (1997) [не прочитана]
print(repr(книга2))     # Книга('Властелин колец', 'Дж. Р. Р. Толкин', 1954)
print(len(книга1))      # 200

print(книга1 == книга2) # False
print(книга1 < книга2)  # False (1997 > 1954)

книга1()  # Читаем книгу: Гарри Поттер
print(книга1)  # Теперь статус "прочитана"

# Базовый класс (родительский)
class ТранспортноеСредство:
    def __init__(self, название, скорость):
        self.название = название
        self.скорость = скорость
        self.в_движении = False
    
    def начать_движение(self):
        if not self.в_движении:
            self.в_движении = True
            print(f"{self.название} начал движение!")
        else:
            print(f"{self.название} уже в движении")
    
    def остановиться(self):
        if self.в_движении:
            self.в_движении = False
            print(f"{self.название} остановился")
        else:
            print(f"{self.название} уже стоит")
    
    def описание(self):
        return f"{self.название} (скорость: {self.скорость} км/ч)"

# Дочерний класс
class Автомобиль(ТранспортноеСредство):
    def __init__(self, название, скорость, марка, топливо):
        # Вызов конструктора родительского класса
        super().__init__(название, скорость)
        # Дополнительные атрибуты
        self.марка = марка
        self.топливо = топливо
        self.уровень_топлива = 100
    
    # Переопределение метода
    def описание(self):
        базовая_инфо = super().описание()
        return f"{базовая_инфо}, марка: {self.марка}, топливо: {self.топливо}"
    
    # Новый метод
    def заправиться(self, количество):
        self.уровень_топлива += количество
        if self.уровень_топлива > 100:
            self.уровень_топлива = 100
        print(f"Заправлено. Уровень топлива: {self.уровень_топлива}%")

# Еще один дочерний класс
class Велосипед(ТранспортноеСредство):
    def __init__(self, название, скорость, тип):
        super().__init__(название, скорость)
        self.тип = тип
        self.передача = 1
    
    def переключить_передачу(self, передача):
        if 1 <= передача <= 7:
            self.передача = передача
            print(f"Переключено на передачу {передача}")
        else:
            print("Недопустимая передача")

# Использование
авто = Автомобиль("Машина", 120, "Toyota", "бензин")
велосипед = Велосипед("Велик", 25, "горный")

print(авто.описание())      # Машина (скорость: 120 км/ч), марка: Toyota...
print(велосипед.описание()) # Велик (скорость: 25 км/ч)

авто.начать_движение()      # Машина начал движение!
авто.заправиться(30)        # Заправлено. Уровень топлива: 100%

велосипед.начать_движение() # Велик начал движение!
велосипед.переключить_передачу(3)  # Переключено на передачу 3

Множественное наследование и миксины

Класс может наследоваться от нескольких родительских классов.

# Миксины - классы, добавляющие определенную функциональность
class Сохраняемый:
    def сохранить(self, имя_файла):
        import json
        данные = self.__dict__  # Все атрибуты объекта
        with open(имя_файла, 'w', encoding='utf-8') as файл:
            json.dump(данные, файл, ensure_ascii=False, indent=2)
        print(f"Объект сохранен в {имя_файла}")
    
    def загрузить(self, имя_файла):
        import json
        with open(имя_файла, 'r', encoding='utf-8') as файл:
            данные = json.load(файл)
        for ключ, значение in данные.items():
            setattr(self, ключ, значение)
        print(f"Объект загружен из {имя_файла}")

class Логируемый:
    def __init__(self):
        self.действия = []
    
    def добавить_действие(self, действие):
        import datetime
        время = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.действия.append(f"{время}: {действие}")
    
    def показать_лог(self):
        print("Лог действий:")
        for действие in self.действия:
            print(f"  {действие}")

# Класс с множественным наследованием
class УмныйТранспорт(ТранспортноеСредство, Сохраняемый, Логируемый):
    def __init__(self, название, скорость):
        # Инициализируем все родительские классы
        ТранспортноеСредство.__init__(self, название, скорость)
        Логируемый.__init__(self)
        # Сохраняемый не требует __init__
    
    def начать_движение(self):
        super().начать_движение()
        self.добавить_действие("Начало движения")
    
    def остановиться(self):
        super().остановиться()
        self.добавить_действие("Остановка")

# Порядок разрешения методов (MRO - Method Resolution Order)
print("Порядок поиска методов:")
for класс in УмныйТранспорт.__mro__:
    print(f"  {класс.__name__}")

# Использование
умная_машина = УмныйТранспорт("Умная машина", 150)
умная_машина.начать_движение()  # Умная машина начал движение!
умная_машина.остановиться()     # Умная машина остановился

умная_машина.показать_лог()
# Лог действий:
#   2023-10-15 14:30:00: Начало движения
#   2023-10-15 14:30:01: Остановка

умная_машина.сохранить("умная_машина.json")
# Объект сохранен в умная_машина.json

# Создание нового объекта и загрузка данных
новая_машина = УмныйТранспорт("Новая", 0)
новая_машина.загрузить("умная_машина.json")
print(f"Загруженная машина: {новая_машина.название}")  # Умная машина

Инкапсуляция и свойства

Сокрытие внутренней реализации и контроль доступа к данным.

class БанковскийСчет:
    def __init__(self, владелец, начальный_баланс=0):
        self.владелец = владелец
        self.__баланс = начальный_баланс  # Приватный атрибут (два подчеркивания)
        self.__история = []  # Приватный список операций
    
    # Геттер (получение значения)
    def получить_баланс(self):
        return self.__баланс
    
    # Сеттер (установка значения с проверкой)
    def установить_баланс(self, сумма):
        if сумма >= 0:
            разница = сумма - self.__баланс
            self.__баланс = сумма
            if разница != 0:
                операция = "пополнение" if разница > 0 else "снятие"
                self.__история.append(f"{операция}: {abs(разница)}")
        else:
            print("Баланс не может быть отрицательным")
    
    # Использование property (более питонический способ)
    @property
    def баланс(self):
        """Геттер через property"""
        return self.__баланс
    
    @баланс.setter
    def баланс(self, значение):
        """Сеттер через property"""
        if значение >= 0:
            старое_значение = self.__баланс
            self.__баланс = значение
            изменение = значение - старое_значение
            if изменение != 0:
                тип = "пополнение" if изменение > 0 else "снятие"
                self.__история.append(f"{тип}: {abs(изменение)}")
        else:
            print("Ошибка: отрицательный баланс")
    
    @property
    def история(self):
        """Только чтение (нет сеттера)"""
        return self.__истории.copy()  # Возвращаем копию
    
    # Методы для работы со счетом
    def пополнить(self, сумма):
        if сумма > 0:
            self.баланс += сумма  # Используем property
            print(f"Счет пополнен на {сумма}")
        else:
            print("Сумма должна быть положительной")
    
    def снять(self, сумма):
        if 0 < сумма <= self.баланс:
            self.баланс -= сумма
            print(f"Со счета снято {сумма}")
        else:
            print("Недостаточно средств или неверная сумма")
    
    def показать_информацию(self):
        print(f"Владелец: {self.владелец}")
        print(f"Баланс: {self.баланс}")
        print(f"История операций: {self.история}")

# Использование
счет = БанковскийСчет("Анна", 1000)

# Прямой доступ к приватным атрибутам невозможен
# print(счет.__баланс)  # Ошибка!

# Используем методы
счет.пополнить(500)     # Счет пополнен на 500
счет.снять(200)         # Со счета снято 200

# Используем property
print(f"Текущий баланс: {счет.баланс}")  # 1300

# Попытка установить отрицательный баланс через property
счет.баланс = -100  # Ошибка: отрицательный баланс

счет.показать_информацию()
# Владелец: Анна
# Баланс: 1300
# История операций: ['пополнение: 500', 'снятие: 200']

Полиморфизм и абстрактные классы

Разные объекты могут иметь одинаковый интерфейс, но разную реализацию.

from abc import ABC, abstractmethod

# Абстрактный базовый класс
class Фигура(ABC):
    def __init__(self, цвет):
        self.цвет = цвет
    
    @abstractmethod
    def площадь(self):
        """Абстрактный метод - должен быть реализован в дочерних классах"""
        pass
    
    @abstractmethod
    def периметр(self):
        pass
    
    def описание(self):
        return f"Фигура цвета {self.цвет}"

# Конкретные классы
class Прямоугольник(Фигура):
    def __init__(self, цвет, длина, ширина):
        super().__init__(цвет)
        self.длина = длина
        self.ширина = ширина
    
    def площадь(self):
        return self.длина * self.ширина
    
    def периметр(self):
        return 2 * (self.длина + self.ширина)
    
    def описание(self):
        return f"Прямоугольник {self.цвет} цвета, {self.длина}x{self.ширина}"

class Круг(Фигура):
    def __init__(self, цвет, радиус):
        super().__init__(цвет)
        self.радиус = радиус
    
    def площадь(self):
        import math
        return math.pi * self.радиус ** 2
    
    def периметр(self):
        import math
        return 2 * math.pi * self.радиус
    
    def описание(self):
        return f"Круг {self.цвет} цвета, радиус {self.радиус}"

class Треугольник(Фигура):
    def __init__(self, цвет, сторона_а, сторона_б, сторона_в):
        super().__init__(цвет)
        self.а = сторона_а
        self.б = сторона_б
        self.в = сторона_в
    
    def площадь(self):
        # Формула Герона
        p = self.периметр() / 2
        return (p * (p - self.а) * (p - self.б) * (p - self.в)) ** 0.5
    
    def периметр(self):
        return self.а + self.б + self.в
    
    def описание(self):
        return f"Треугольник {self.цвет} цвета, стороны: {self.а}, {self.б}, {self.в}"

# Полиморфизм в действии
фигуры = [
    Прямоугольник("красный", 10, 5),
    Круг("синий", 7),
    Треугольник("зеленый", 3, 4, 5)
]

# Одинаковый интерфейс для разных объектов
for фигура in фигуры:
    print(f"{фигура.описание()}")
    print(f"  Площадь: {фигура.площадь():.2f}")
    print(f"  Периметр: {фигура.периметр():.2f}")
    print()

# Статический и классовый методы
class Математика:
    # Статический метод - не требует self или cls
    @staticmethod
    def сложить(a, b):
        return a + b
    
    # Классовый метод - получает класс как первый аргумент
    @classmethod
    def создать_из_строки(cls, строка):
        """Создает объект из строки 'a,b'"""
        a, b = map(int, строка.split(','))
        return cls.сложить(a, b)  # Используем статический метод

# Использование
print(Математика.сложить(5, 3))  # 8
print(Математика.создать_из_строки("10,20"))  # 30

# Можно вызывать и через объект
математика = Математика()
print(математика.сложить(7, 8))  # 15

from abc import ABC, abstractmethod
from datetime import datetime, timedelta

class БиблиотечныйЭкземпляр(ABC):
    def __init__(self, название, автор, год, идентификатор):
        self.__идентификатор = идентификатор
        self.название = название
        self.автор = автор
        self.год = год
        self.выдано_читателю = None
        self.дата_выдачи = None
    
    @abstractmethod
    def получить_тип(self):
        pass
    
    @property
    def доступна(self):
        return self.выдано_читателю is None
    
    def выдать(self, читатель):
        if self.доступна:
            self.выдано_читателю = читатель
            self.дата_выдачи = datetime.now()
            return True
        return False
    
    def вернуть(self):
        self.выдано_читателю = None
        self.дата_выдачи = None
    
    def __str__(self):
        статус = "доступна" if self.доступна else f"выдана {self.выдано_читателю}"
        return f"{self.название} ({self.автор}, {self.год}) - {статус}"

class Книга(БиблиотечныйЭкземпляр):
    def __init__(self, название, автор, год, идентификатор, жанр, страницы):
        super().__init__(название, автор, год, идентификатор)
        self.жанр = жанр
        self.страницы = страницы
    
    def получить_тип(self):
        return "Книга"
    
    def информация(self):
        return f"{super().__str__()}, жанр: {self.жанр}, {self.страницы} стр."
