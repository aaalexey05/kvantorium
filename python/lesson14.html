Урок 14: Trinket - Онлайн среда Python

Программируем прямо в браузере без установки

Что такое Trinket?

Trinket — это онлайн-платформа, которая позволяет писать и запускать код Python прямо в браузере. Не нужно ничего устанавливать!

Работа в браузере

Не требует установки, работает на любом устройстве

Совместная работа

Редактируйте код вместе с друзьями в реальном времени

Автосохранение

Код автоматически сохраняется в облаке

Идеально для обучения

Встроенные уроки и примеры кода

# Trinket позволяет запускать такой код прямо в браузере:
# Попробуйте изменить числа и нажмите "Run"

# Простая программа-калькулятор
def калькулятор():
    print("=== ПРОСТОЙ КАЛЬКУЛЯТОР ===")
    
    try:
        a = float(input("Введите первое число: "))
        операция = input("Выберите операцию (+, -, *, /): ")
        b = float(input("Введите второе число: "))
        
        if операция == '+':
            результат = a + b
        elif операция == '-':
            результат = a - b
        elif операция == '*':
            результат = a * b
        elif операция == '/':
            if b != 0:
                результат = a / b
            else:
                print("Ошибка: деление на ноль!")
                return
        else:
            print("Неверная операция!")
            return
        
        print(f"Результат: {a} {операция} {b} = {результат}")
        
    except ValueError:
        print("Пожалуйста, вводите числа!")

# Запускаем калькулятор
if __name__ == "__main__":
    калькулятор()

def draw_spiral():
    t.penup()
    t.goto(0, 0)
    t.pendown()
    
    for i in range(100):
        t.color(
            random.random(),  # Красный
            random.random(),  # Зеленый
            random.random()   # Синий
        )
        t.forward(i * 2)
        t.right(91)

# Функция для рисования звездного неба
def draw_stars():
    star = turtle.Turtle()
    star.speed(0)
    star.color("white")
    star.penup()
    star.hideturtle()
    
    for _ in range(50):
        x = random.randint(-300, 300)
        y = random.randint(-200, 200)
        size = random.randint(1, 3)
        
        star.goto(x, y)
        star.dot(size, "white")

# Функция для анимации
def animate():
    colors = ["red", "orange", "yellow", "green", "blue", "purple"]
    
    for color in colors:
        t.color(color)
        t.circle(50)
        t.right(60)
        time.sleep(0.5)

# Файл: utils.py (вспомогательные функции)

Модуль с вспомогательными функциями

def калькулятор(a, b, операция):
    """Выполняет математические операции"""
    if операция == '+':
        return a + b
    elif операция == '-':
        return a - b
    elif операция == '*':
        return a * b
    elif операция == '/':
        if b != 0:
            return a / b
        else:
            return "Ошибка: деление на ноль!"
    else:
        return "Неизвестная операция"

def проверка_числа(число):
    """Проверяет, является ли число четным"""
    return число % 2 == 0

# Файл: data.py (данные)
"""
Модуль с данными

студенты = [
    {"имя": "Анна", "возраст": 12, "оценка": 5},
    {"имя": "Иван", "возраст": 13, "оценка": 4},
    {"имя": "Мария", "возраст": 12, "оценка": 5},
    {"имя": "Сергей", "возраст": 13, "оценка": 3},
    {"имя": "Ольга", "возраст": 12, "оценка": 4}
]

предметы = ["Математика", "Физика", "Информатика", "Русский язык", "История"]

# Файл: main.py (основная программа)
"""
Основная программа
"""

# Импортируем наши модули
import utils
import data

def главное_меню():
    """Основное меню программы"""
    print("=" * 50)
    print("ГЛАВНОЕ МЕНЮ")
    print("=" * 50)
    print("1. Приветствие")
    print("2. Калькулятор")
    print("3. Проверка числа")
    print("4. Список студентов")
    print("5. Выход")

def показать_студентов():
    """Показывает список студентов"""
    print("\nСписок студентов:")
    print("-" * 30)
    for студент in data.студенты:
        print(f"Имя: {студент['имя']}")
        print(f"Возраст: {студент['возраст']}")
        print(f"Оценка: {студент['оценка']}")

# - Можно отключать/включать участников

# Пример проекта для совместной работы:

# Файл: game_plan.md (план проекта)
"""
ПЛАН РАЗРАБОТКИ ИГРЫ
====================

Название: "Космическое приключение"

Участники:
1. [Имя] - игровая логика
2. [Имя] - графика и анимации
3. [Имя] - звуки и музыка
4. [Имя] - уровни и дизайн

Задачи:
- Неделя 1: Основная механика
- Неделя 2: Графика и анимации
- Неделя 3: Звуки и музыка
- Неделя 4: Тестирование и улучшения

Технологии:
- Python
- Pygame (для графики)
- Trinket (для совместной работы)
"""

# Файл: main.py (основной файл игры)
import pygame
import random
import sys

# Инициализация Pygame
pygame.init()

# Настройки экрана
WIDTH, HEIGHT = 800, 600
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Космическое приключение")

# Цвета
BLACK = (0, 0, 0)
WHITE = (255, 255, 255)
GREEN = (0, 255, 0)
RED = (255, 0, 0)
BLUE = (0, 120, 255)
YELLOW = (255, 255, 0)

# Класс игрока
class Player:
    def __init__(self):
        self.x = WIDTH // 2
        self.y = HEIGHT - 100
        self.width = 50
        self.height = 50
        self.speed = 5
        self.color = GREEN
        self.health = 100
        self.score = 0
    
    def draw(self):
        # Рисуем корабль игрока
        pygame.draw.rect(screen, self.color, 
                        (self.x - self.width//2, self.y - self.height//2, 
                         self.width, self.height))
        
        # Рисуем "нос" корабля
        pygame.draw.polygon(screen, BLUE, [
            (self.x, self.y - self.height//2 - 20),
            (self.x - 15, self.y - self.height//2),
            (self.x + 15, self.y - self.height//2)
        ])
    
    def move(self, keys):
        if keys[pygame.K_LEFT] and self.x > self.width//2:
            self.x -= self.speed
        if keys[pygame.K_RIGHT] and self.x < WIDTH - self.width//2:
            self.x += self.speed
        if keys[pygame.K_UP] and self.y > self.height//2:
            self.y -= self.speed
        if keys[pygame.K_DOWN] and self.y < HEIGHT - self.height//2:
            self.y += self.speed

# Класс врага
class Enemy:
    def __init__(self):
        self.x = random.randint(50, WIDTH - 50)
        self.y = random.randint(-100, -50)
        self.width = 40
        self.height = 40
        self.speed = random.uniform(1, 3)
        self.color = RED
    
    def update(self):
        self.y += self.speed
        return self.y < HEIGHT + 50
    
    def draw(self):
        pygame.draw.rect(screen, self.color,
                        (self.x - self.width//2, self.y - self.height//2,
                         self.width, self.height))
        
        # Рисуем "глаза" врага
        pygame.draw.circle(screen, YELLOW, 
                          (self.x - 10, self.y - 5), 5)
        pygame.draw.circle(screen, YELLOW,
                          (self.x + 10, self.y - 5), 5)

# Основная функция игры
def main_game():
    clock = pygame.time.Clock()
    player = Player()
    enemies = []
    spawn_timer = 0
    font = pygame.font.Font(None, 36)
    
    running = True
    while running:
        # Обработка событий
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_ESCAPE:
                    running = False
        
        # Движение игрока
        keys = pygame.key_pressed()
        player.move(keys)
        
        # Спавн врагов
        spawn_timer += 1
        if spawn_timer >= 60:  # Каждую секунду
            enemies.append(Enemy())
            spawn_timer = 0
        
        # Обновление врагов
        enemies = [enemy for enemy in enemies if enemy.update()]
        
        # Очистка экрана
        screen.fill(BLACK)
        
        # Рисуем звездное небо
        for _ in range(100):
            x = random.randint(0, WIDTH)
            y = random.randint(0, HEIGHT)
            pygame.draw.circle(screen, WHITE, (x, y), 1)
        
        # Рисуем игрока и врагов
        player.draw()
        for enemy in enemies:
            enemy.draw()
        
        # Отображение счета и здоровья
        score_text = font.render(f"Счет: {player.score}", True, WHITE)
        health_text = font.render(f"Здоровье: {player.health}", True, WHITE)
        
        screen.blit(score_text, (10, 10))
        screen.blit(health_text, (10, 50))
        
        # Обновление экрана
        pygame.display.flip()
        clock.tick(60)
    
    pygame.quit()
    sys.exit()

# Запуск игры
if __name__ == "__main__":
    main_game()

            "искал потерянный артефакт", "спасал принцессу", "разгадывал тайну",
            "учился магии", "строил робота", "путешествовал во времени"
        ]
        
        self.events = [
            "столкнулся с драконом", "нашел карту сокровищ", "получил волшебный предмет",
            "встретил старого друга", "попал в ловушку", "открыл портал в другое измерение"
        ]
        
        self.endings = [
            "и жил долго и счастливо", "стал правителем королевства",
            "нашел свое истинное призвание", "вернулся домой героем",
            "продолжил свои приключения", "написал книгу о своих путешествиях"
        ]
    
    def generate_story(self):
        """Генерирует случайную историю"""
        character = random.choice(self.characters)
        place = random.choice(self.places)
        action = random.choice(self.actions)
        event = random.choice(self.events)
        ending = random.choice(self.endings)
        
        story = f"""
        🏰 СЛУЧАЙНАЯ ИСТОРИЯ 🏰
        
        Жил-был {character}, который жил {place}.
        Однажды он {action}.
        
        Во время своего путешествия он {event}.
        После многих испытаний и приключений он {ending}.
        
        КОНЕЦ.
        """
        
        return story
    
    def interactive_story(self):
        """Интерактивная история с выбором пользователя"""
        print("🎭 СОЗДАЙ СВОЮ ИСТОРИЮ 🎭")
        print("=" * 40)
        
        # Выбор персонажа
        print("\nВыбери главного героя:")
        for i, char in enumerate(self.characters, 1):
            print(f"{i}. {char}")
        
        char_choice = int(input("\nТвой выбор (1-6): ")) - 1
        character = self.characters[char_choice]
        
        # Выбор места
        print("\nВыбери место действия:")
        for i, place in enumerate(self.places, 1):
            print(f"{i}. {place}")
        
        place_choice = int(input("\nТвой выбор (1-6): ")) - 1
        place = self.places[place_choice]
        
        # Выбор действия
        print("\nВыбери, что делает герой:")
        for i, action in enumerate(self.actions, 1):
            print(f"{i}. {action}")
        
        action_choice = int(input("\nТвой выбор (1-6): ")) - 1
        action = self.actions[action_choice]
        
        # Генерация финала
        print("\n✨ ГЕНЕРИРУЕМ ИСТОРИЮ... ✨")
        time.sleep(2)
        
        event = random.choice(self.events)
        ending = random.choice(self.endings)
        
        story = f"""
        📖 ТВОЯ ИСТОРИЯ 📖
        
        Жил-был {character}, который жил {place}.
        Однажды он решил {action}.
        
        Во время своего приключения он {event}.
        В конце концов он {ending}.
        
        Надеемся, тебе понравилась эта история!
        """
        
        return story
    
    def save_story(self, story, filename="моя_история.txt"):
        """Сохраняет историю в файл (в Trinket это создаст файл для скачивания)"""
        # В Trinket мы можем создать текстовый файл
        with open(filename, "w", encoding="utf-8") as f:
            f.write(story)
        
        print(f"\n📄 История сохранена в файл: {filename}")
        print("В Trinket файл появится в списке файлов слева")
