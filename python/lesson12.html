# game/player.py - Модуль игрока
import json
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from enum import Enum

class PlayerClass(Enum):
    FRONTEND = "Фронтенд-воин"
    BACKEND = "Бэкенд-маг"
    FULLSTACK = "Фуллстек-универсал"

@dataclass
class Item:
    name: str
    description: str
    value: int
    item_type: str  # weapon, armor, consumable, key
    
    def __str__(self):
        return f"{self.name} ({self.description})"

@dataclass
class Player:
    name: str
    player_class: PlayerClass
    level: int = 1
    experience: int = 0
    health: int = 100
    max_health: int = 100
    mana: int = 50
    max_mana: int = 50
    attack: int = 10
    defense: int = 5
    inventory: List[Item] = field(default_factory=list)
    equipped: Dict[str, Optional[Item]] = field(default_factory=lambda: {
        'weapon': None, 'armor': None, 'accessory': None
    })
    location: str = "start_town"
    
    @property
    def required_experience(self):
        return self.level * 100
    
    def gain_experience(self, amount):
        self.experience += amount
        print(f"Получено {amount} опыта!")
        
        while self.experience >= self.required_experience:
            self.level_up()
    
    def level_up(self):
        self.level += 1
        self.experience -= self.required_experience
        self.max_health += 20
        self.max_mana += 10
        self.attack += 5
        self.defense += 3
        
        self.health = self.max_health
        self.mana = self.max_mana
        
        print(f"🎉 Уровень повышен! Теперь вы {self.level} уровня!")
        print(f"❤️ Здоровье: {self.health}/{self.max_health}")
        print(f"🔮 Мана: {self.mana}/{self.max_mana}")
    
    def take_damage(self, damage):
        actual_damage = max(1, damage - self.defense)
        self.health -= actual_damage
        return actual_damage
    
    def heal(self, amount):
        self.health = min(self.max_health, self.health + amount)
    
    def add_item(self, item):
        self.inventory.append(item)
        print(f"В инвентарь добавлен: {item}")
    
    def equip_item(self, item):
        if item.item_type in self.equipped:
            # Сначала снимаем текущий предмет
            if self.equipped[item.item_type]:
                self.unequip_item(item.item_type)
            
            self.equipped[item.item_type] = item
            print(f"Экипирован: {item.name}")
            
            # Применяем бонусы
            if item.item_type == 'weapon':
                self.attack += item.value
            elif item.item_type == 'armor':
                self.defense += item.value
        else:
            print(f"Нельзя экипировать предмет типа {item.item_type}")
    
    def unequip_item(self, slot):
        if self.equipped[slot]:
            item = self.equipped[slot]
            self.equipped[slot] = None
            
            # Убираем бонусы
            if slot == 'weapon':
                self.attack -= item.value
            elif slot == 'armor':
                self.defense -= item.value
            
            print(f"Снят: {item.name}")
            return item
        return None
    
    def to_dict(self):
        """Конвертирует игрока в словарь для сохранения"""
        return {
            'name': self.name,
            'player_class': self.player_class.value,
            'level': self.level,
            'experience': self.experience,
            'health': self.health,
            'max_health': self.max_health,
            'mana': self.mana,
            'max_mana': self.max_mana,
            'attack': self.attack,
            'defense': self.defense,
            'inventory': [
                {'name': i.name, 'description': i.description, 
                 'value': i.value, 'item_type': i.item_type}
                for i in self.inventory
            ],
            'equipped': {
                slot: (item.to_dict() if item else None)
                for slot, item in self.equipped.items()
            },
            'location': self.location
        }
    
    @classmethod
    def from_dict(cls, data):
        """Создает игрока из словаря"""
        player = cls(
            name=data['name'],
            player_class=PlayerClass(data['player_class'])
        )
        
        # Восстанавливаем остальные атрибуты
        for attr in ['level', 'experience', 'health', 'max_health', 
                    'mana', 'max_mana', 'attack', 'defense', 'location']:
            setattr(player, attr, data[attr])
        
        # Восстанавливаем инвентарь
        player.inventory = [
            Item(**item_data) for item_data in data['inventory']
        ]
        
        # Восстанавливаем экипировку
        for slot, item_data in data['equipped'].items():
            if item_data:
                player.equipped[slot] = Item(**item_data)
        
        return player

# game/combat.py - Боевая система
import random
from typing import List, Tuple
from enum import Enum
from game.player import Player

class MoveType(Enum):
    ATTACK = "атака"
    DEFENSE = "защита"
    HEAL = "лечение"
    SPECIAL = "особое"

class Move:
    def __init__(self, name: str, move_type: MoveType, 
                 power: int, mana_cost: int, description: str):
        self.name = name
        self.move_type = move_type
        self.power = power
        self.mana_cost = mana_cost
        self.description = description
    
    def __str__(self):
        return f"{self.name} ({self.move_type.value}, сила: {self.power}, мана: {self.mana_cost})"

class Enemy:
    def __init__(self, name: str, health: int, attack: int, 
                 defense: int, moves: List[Move], xp_reward: int):
        self.name = name
        self.max_health = health
        self.health = health
        self.attack = attack
        self.defense = defense
        self.moves = moves
        self.xp_reward = xp_reward
    
    def is_alive(self):
        return self.health > 0
    
    def take_damage(self, damage):
        actual_damage = max(1, damage - self.defense)
        self.health -= actual_damage
        return actual_damage
    
    def choose_move(self):
        return random.choice(self.moves)

class Combat:
    def __init__(self, player: Player, enemy: Enemy):
        self.player = player
        self.enemy = enemy
        self.turn = 0
        self.player_moves = self.get_player_moves()
    
    def get_player_moves(self):
        """Возвращает доступные ходы игрока в зависимости от класса"""
        base_moves = [
            Move("Базовый удар", MoveType.ATTACK, 10, 0, "Базовая атака"),
            Move("Защитная стойка", MoveType.DEFENSE, 0, 5, "Увеличивает защиту"),
            Move("Лечение", MoveType.HEAL, 20, 10, "Восстанавливает здоровье"),
        ]
        
        # Классовые умения
        if self.player.player_class.value == "Фронтенд-воин":
            class_moves = [
                Move("HTML-удар", MoveType.SPECIAL, 25, 15, "Мощный удар тегами"),
                Move("CSS-щит", MoveType.DEFENSE, 0, 10, "Создает защитный щит"),
            ]
        elif self.player.player_class.value == "Бэкенд-маг":
            class_moves = [
                Move("Python-огонь", MoveType.SPECIAL, 30, 20, "Огненная атака Python"),
                Move("База данных", MoveType.HEAL, 40, 25, "Восстановление из БД"),
            ]
        else:  # Фуллстек
            class_moves = [
                Move("Fullstack комбо", MoveType.SPECIAL, 35, 30, "Комбинированная атака"),
                Move("API-исцеление", MoveType.HEAL, 30, 20, "Лечение через API"),
            ]
        
        return base_moves + class_moves
    
    def player_turn(self, move_index: int) -> Tuple[str, bool]:
        """Ход игрока, возвращает результат и флаг окончания боя"""
        if move_index >= len(self.player_moves):
            return "Неверный ход", False
        
        move = self.player_moves[move_index]
        
        # Проверка маны
        if self.player.mana < move.mana_cost:
            return f"Недостаточно маны для {move.name}!", False
        
        self.player.mana -= move.mana_cost
        result = ""
        
        if move.move_type == MoveType.ATTACK:
            damage = move.power + self.player.attack
            actual_damage = self.enemy.take_damage(damage)
            result = f"Вы использовали {move.name} и нанесли {actual_damage} урона!"
        
        elif move.move_type == MoveType.DEFENSE:
            self.player.defense += move.power
            result = f"Ваша защита увеличена на {move.power}!"
        
        elif move.move_type == MoveType.HEAL:
            heal_amount = move.power
            self.player.heal(heal_amount)
            result = f"Вы восстановили {heal_amount} здоровья!"
        
        elif move.move_type == MoveType.SPECIAL:
            damage = move.power + self.player.attack * 2
            actual_damage = self.enemy.take_damage(damage)
            result = f"Особое умение {move.name} нанесло {actual_damage} урона!"
        
        # Проверяем победу
        if not self.enemy.is_alive():
            xp = self.enemy.xp_reward
            self.player.gain_experience(xp)
            return f"{result}\n🏆 Вы победили {self.enemy.name}! Получено {xp} опыта.", True
        
        return result, False
    
    def enemy_turn(self) -> Tuple[str, bool]:
        """Ход врага, возвращает результат и флаг поражения"""
        move = self.enemy.choose_move()
        result = ""
        
        if move.move_type == MoveType.ATTACK:
            damage = move.power + self.enemy.attack
            actual_damage = self.player.take_damage(damage)
            result = f"{self.enemy.name} использует {move.name} и наносит {actual_damage} урона!"
        
        elif move.move_type == MoveType.DEFENSE:
            self.enemy.defense += move.power
            result = f"{self.enemy.name} усиливает защиту на {move.power}!"
        
        elif move.move_type == MoveType.HEAL:
            heal_amount = min(move.power, self.enemy.max_health - self.enemy.health)
            self.enemy.health += heal_amount
            result = f"{self.enemy.name} восстанавливает {heal_amount} здоровья!"
        
        # Проверяем поражение
        if self.player.health <= 0:
            return f"{result}\n💀 Вы проиграли...", True
        
        return result, False
    
    def get_status(self):
        """Возвращает текущий статус боя"""
        player_hp_percent = (self.player.health / self.player.max_health) * 100
        enemy_hp_percent = (self.enemy.health / self.enemy.max_health) * 100
        
        status = f"""
{'='*50}
ИГРОК: {self.player.name} ({self.player.player_class.value})
❤️ Здоровье: {self.player.health}/{self.player.max_health} [{self.get_bar(player_hp_percent)}]
🔮 Мана: {self.player.mana}/{self.player.max_mana}

ПРОТИВНИК: {self.enemy.name}
❤️ Здоровье: {self.enemy.health}/{self.enemy.max_health} [{self.get_bar(enemy_hp_percent)}]
{'='*50}
        """
        return status
    
    @staticmethod
    def get_bar(percent, length=20):
        filled = int(percent * length / 100)
        return "█" * filled + "░" * (length - filled)

# game/locations.py - Локации и перемещение
import json
from typing import Dict, List, Optional
from dataclasses import dataclass
from game.combat import Enemy, Move, MoveType

@dataclass
class Location:
    name: str
    description: str
    exits: Dict[str, str]  # направление -> название_локации
    npcs: List[str]
    enemies: List[Enemy]
    items: List[str]
    quests: List[str]
    is_safe: bool = True  # Можно ли сохраняться здесь

class World:
    def __init__(self, data_file="data/locations.json"):
        self.locations: Dict[str, Location] = {}
        self.current_location = "start_town"
        self.load_world(data_file)
    
    def load_world(self, data_file):
        """Загружает мир из JSON файла"""
        with open(data_file, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        for loc_name, loc_data in data.items():
            # Создаем врагов для локации
            enemies = []
            for enemy_data in loc_data.get('enemies', []):
                moves = []
                for move_data in enemy_data.get('moves', []):
                    move = Move(
                        name=move_data['name'],
                        move_type=MoveType(move_data['type']),
                        power=move_data['power'],
                        mana_cost=move_data.get('mana_cost', 0),
                        description=move_data['description']
                    )
                    moves.append(move)
                
                enemy = Enemy(
                    name=enemy_data['name'],
                    health=enemy_data['health'],
                    attack=enemy_data['attack'],
                    defense=enemy_data['defense'],
                    moves=moves,
                    xp_reward=enemy_data['xp_reward']
                )
                enemies.append(enemy)
            
            location = Location(
                name=loc_data['name'],
                description=loc_data['description'],
                exits=loc_data.get('exits', {}),
                npcs=loc_data.get('npcs', []),
                enemies=enemies,
                items=loc_data.get('items', []),
                quests=loc_data.get('quests', []),
                is_safe=loc_data.get('is_safe', True)
            )
            self.locations[loc_name] = location
    
    def move(self, direction: str) -> Optional[str]:
        """Перемещает игрока в указанном направлении"""
        current = self.locations[self.current_location]
        
        if direction not in current.exits:
            return f"Нельзя пойти {direction}"
        
        new_location = current.exits[direction]
        if new_location not in self.locations:
            return f"Локация {new_location} не найдена"
        
        self.current_location = new_location
        new_loc = self.locations[new_location]
        
        # Проверяем случайную встречу с врагом
        if not new_loc.is_safe and new_loc.enemies:
            encounter_chance = random.random()
            if encounter_chance > 0.7:  # 30% шанс встречи
                enemy = random.choice(new_loc.enemies)
                return f"Вы встретили {enemy.name}! Приготовьтесь к бою!"
        
        return None  # Нет ошибок и случайных встреч
    
    def get_current_location(self) -> Location:
        return self.locations[self.current_location]
    
    def get_available_directions(self):
        current = self.locations[self.current_location]
        return list(current.exits.keys())
    
    def describe_current_location(self):
        loc = self.get_current_location()
        description = f"""
📍 {loc.name}
{loc.description}

Выходы: {', '.join(loc.exits.keys()) if loc.exits else 'нет'}
"""
        
        if loc.npcs:
            description += f"\nЗдесь находятся: {', '.join(loc.npcs)}"
        
        if loc.items:
            description += f"\nПредметы: {', '.join(loc.items)}"
        
        if loc.quests:
            description += f"\nДоступные квесты: {', '.join(loc.quests)}"
        
        return description

# game/quests.py - Система квестов
class QuestStatus(Enum):
    NOT_STARTED = "не начат"
    IN_PROGRESS = "в процессе"
    COMPLETED = "выполнен"
    FAILED = "провален"

class Quest:
    def __init__(self, quest_id: str, name: str, description: str,
                 requirements: Dict, rewards: Dict, steps: List[Dict]):
        self.quest_id = quest_id
        self.name = name
        self.description = description
        self.requirements = requirements  # уровень, предметы и т.д.
        self.rewards = rewards  # опыт, предметы, золото
        self.steps = steps
        self.current_step = 0
        self.status = QuestStatus.NOT_STARTED
    
    def can_start(self, player):
        """Проверяет, может ли игрок начать квест"""
        if 'min_level' in self.requirements:
            if player.level < self.requirements['min_level']:
                return False
        
        if 'required_quests' in self.requirements:
            # Проверяем выполненные квесты
            pass
        
        return True
    
    def start(self):
        if self.status == QuestStatus.NOT_STARTED:
            self.status = QuestStatus.IN_PROGRESS
            return f"Квест '{self.name}' начат!"
        return "Нельзя начать этот квест"
    
    def update_progress(self, event_type: str, event_data: Dict):
        """Обновляет прогресс квеста на основе событий"""
        if self.status != QuestStatus.IN_PROGRESS:
            return
        
        current_step = self.steps[self.current_step]
        
        if current_step['type'] == event_type:
            # Проверяем условие шага
            if self.check_step_condition(current_step, event_data):
                self.current_step += 1
                
                if self.current_step >= len(self.steps):
                    self.complete()
                    return "Квест выполнен!"
                else:
                    return f"Шаг {self.current_step}/{len(self.steps)} выполнен!"
        
        return None
    
    def check_step_condition(self, step, event_data):
        """Проверяет условие шага квеста"""
        condition = step['condition']
        
        if step['type'] == 'kill':
            return event_data.get('enemy_name') == condition['enemy_name']
        
        elif step['type'] == 'collect':
            return event_data.get('item_name') == condition['item_name']
        
        elif step['type'] == 'visit':
            return event_data.get('location') == condition['location']
        
        return False
    
    def complete(self):
        self.status = QuestStatus.COMPLETED
        return self.rewards
    
    def get_current_step_description(self):
        if self.status == QuestStatus.IN_PROGRESS:
            step = self.steps[self.current_step]
            return f"Текущая цель: {step['description']}"
        return "Квест не активен"

import os

from colorama import init, Fore, Back, Style

init(autoreset=True)  # Инициализация colorama

class Display:
    def __init__(self):
        self.width = 80
        self.colors = {
            'title': Fore.CYAN + Style.BRIGHT,
            'success': Fore.GREEN,
            'error': Fore.RED,
            'warning': Fore.YELLOW,
            'info': Fore.BLUE,
            'player': Fore.MAGENTA,
            'enemy': Fore.RED,
            'item': Fore.YELLOW,
            'location': Fore.GREEN,
        }
    
    def clear_screen(self):
        """Очищает экран консоли"""
        os.system('cls' if os.name == 'nt' else 'clear')
    
    def show_title(self, title):
        """Показывает красивый заголовок"""
        print(self.colors['title'])
        print("=" * self.width)
        print(title.center(self.width))
        print("=" * self.width)
        print(Style.RESET_ALL)
    
    def show_player_status(self, player):
        """Показывает статус игрока"""
        hp_percent = (player.health / player.max_health) * 100
        mana_percent = (player.mana / player.max_mana) * 100
        
        print(f"\n{self.colors['player']}{player.name} - Уровень {player.level}")
        print(f"Класс: {player.player_class.value}")
        
        print(f"❤️  Здоровье: {player.health}/{player.max_health} ", end="")
        print(self.get_health_bar(hp_percent))
        
        print(f"🔮 Мана: {player.mana}/{player.max_mana} ", end="")
        print(self.get_mana_bar(mana_percent))
        
        print(f"⚔️  Атака: {player.attack} 🛡️ Защита: {player.defense}")
        print(f"📈 Опыт: {player.experience}/{player.required_experience}")
        print(Style.RESET_ALL)
    
    def show_location(self, location):
        """Показывает информацию о локации"""
        print(f"\n{self.colors['location']}📍 {location.name}")
        print("─" * len(location.name) * 2)
        print(location.description)
        print(Style.RESET_ALL)
        
        if location.exits:
            exits_str = "Выходы: " + ", ".join(location.exits.keys())
            print(f"\n{exits_str}")
    
    def get_health_bar(self, percent, length=20):
        """Возвращает строку с полоской здоровья"""
        filled = int(percent * length / 100)
        color = Fore.GREEN if percent > 50 else Fore.YELLOW if percent > 20 else Fore.RED
        return color + "█" * filled + "░" * (length - filled) + Style.RESET_ALL
    
    def get_mana_bar(self, percent, length=20):
        """Возвращает строку с полоской маны"""
        filled = int(percent * length / 100)
        color = Fore.BLUE
        return color + "█" * filled + "░" * (length - filled) + Style.RESET_ALL
    
    def typewriter_effect(self, text, delay=0.03):
        """Эффект печатной машинки"""
        for char in text:
            sys.stdout.write(char)
            sys.stdout.flush()
            time.sleep(delay)
        print()
    
    def show_dialogue(self, speaker, text):
        """Показывает диалог с эффектом"""
        print(f"\n{Fore.CYAN}{speaker}:{Style.RESET_ALL}")
        self.typewriter_effect(text)

# utils/input_handler.py - Обработка пользовательского ввода
class InputHandler:
    def get_menu_choice(self, min_val, max_val):
        """Получает выбор из меню с проверкой"""
        while True:
            try:
                choice = input(f"\nВыберите вариант ({min_val}-{max_val}): ")
                if not choice.strip():
                    continue
                
                choice = int(choice)
                if min_val <= choice <= max_val:
                    return choice
                else:
                    print(f"Пожалуйста, введите число от {min_val} до {max_val}")
            except ValueError:
                print("Пожалуйста, введите число")
    
    def get_string_input(self, prompt, min_length=1, max_length=20):
        """Получает строковый ввод с проверкой"""
        while True:
            value = input(prompt).strip()
            
            if len(value) < min_length:
                print(f"Введите не менее {min_length} символов")
            elif len(value) > max_length:
                print(f"Введите не более {max_length} символов")
            else:
                return value
    
    def get_yes_no(self, question):
        """Получает ответ да/нет"""
        while True:
            answer = input(f"{question} (да/нет): ").strip().lower()
            if answer in ['да', 'д', 'yes', 'y']:
                return True
            elif answer in ['нет', 'н', 'no', 'n']:
                return False
            else:
                print("Пожалуйста, ответьте 'да' или 'нет'")

# game/save_system.py - Система сохранения
import json
import os
from datetime import datetime
from game.player import Player

class SaveSystem:
    def __init__(self, save_dir="saves"):
        self.save_dir = save_dir
        if not os.path.exists(save_dir):
            os.makedirs(save_dir)
    
    def save_game(self, player, save_name=None):
        """Сохраняет игру"""
        try:
            if not save_name:
                save_name = f"save_{player.name}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
            
            save_data = {
                'player': player.to_dict(),
                'metadata': {
                    'save_name': save_name,
                    'save_date': datetime.now().isoformat(),
                    'game_version': '1.0',
                    'play_time': 0  # Можно добавить отслеживание времени
                }
            }
            
            save_path = os.path.join(self.save_dir, f"{save_name}.json")
            
            with open(save_path, 'w', encoding='utf-8') as f:
                json.dump(save_data, f, ensure_ascii=False, indent=2)
            
            return True
        except Exception as e:
            print(f"Ошибка сохранения: {e}")
            return False
    
    def load_game(self, save_name):
        """Загружает игру"""
        try:
            save_path = os.path.join(self.save_dir, f"{save_name}.json")
            
            if not os.path.exists(save_path):
                print(f"Сохранение '{save_name}' не найдено")
                return None
            
            with open(save_path, 'r', encoding='utf-8') as f:
                save_data = json.load(f)
            
            player_data = save_data['player']
            player = Player.from_dict(player_data)
            
            print(f"Загружено сохранение от {save_data['metadata']['save_date']}")
            return player
        except Exception as e:
            print(f"Ошибка загрузки: {e}")
            return None
    
    def list_saves(self):
        """Возвращает список доступных сохранений"""
        saves = []
        
        if not os.path.exists(self.save_dir):
            return saves
        
        for filename in os.listdir(self.save_dir):
            if filename.endswith('.json'):
                save_name = filename[:-5]  # Убираем .json
                saves.append(save_name)
        
        return sorted(saves)

Основные улучшения:

  1. Добавьте систему крафта:
    • Рецепты для создания предметов
    • Станки и рабочие места
    • Ремесленные навыки
  2. Реализуйте экономику:
    • Валюту (биткоины, звездочки кода)
    • Торговцев с разными ценами
    • Аукцион или биржу
  3. Добавьте мультиплеер:
    • PvP бои
    • Гильдии и кланы
    • Общие подземелья
  4. Создайте систему достижений:
    • Значки за выполнение заданий
    • Статистика игрока
    • Лидерборды
  5. Добавьте генерацию мира:
    • Процедурные подземелья
    • Случайные события
    • Динамический мир

Технические улучшения:

  1. Добавьте графический интерфейс:
    • Используйте Pygame или tkinter
    • Создайте спрайты для персонажей
    • Добавьте анимации
  2. Реализуйте сетевую игру:
    • Клиент-серверную архитектуру
    • Чат между игроками
    • Облачные сохранения
  3. Создайте инсталлятор:
    • Установочный файл для Windows
    • Пакет для Linux (deb/rpm)
    • Приложение для мобильных
  4. Добавьте моддинг:
    • API для модов
    • Редактор уровней
    • Мастерскую Steam (опционально)

План выполнения:

Неделя 1: Планирование и проектирование
- Создайте дорожную карту проекта
- Нарисуйте архитектуру системы
- Создайте прототип основных механик

Неделя 2: Базовый функционал
- Реализуйте систему крафта
- Добавьте экономику
- Создайте тесты

Неделя 3: Продвинутые функции
- Добавьте мультиплеер
- Реализуйте систему достижений
- Создайте процедурную генерацию

Неделя 4: Полировка и выпуск
- Добавьте графический интерфейс
- Напишите документацию
- Создайте инсталлятор
- Протестируйте и выпустите игру

Критерии оценки:

  • Работоспособность всех основных функций
  • Качество кода (PEP 8, документация, тесты)
  • Пользовательский опыт (интерфейс, удобство)
  • Креативность и оригинальность
  • Сложность реализованных механик
  • Документация и инструкции

Примеры для вдохновения:

  • Dwarf Fortress - сложная процедурная генерация
  • MUD-игры - текстовые многопользовательские игры
  • Rogue - родоначальник жанра roguelike
  • Pokemon - система покемонов и эволюций
  • Minecraft - система крафта и строительства

Советы по разработке игр

  • Начинайте с минимального рабочего прототипа
  • Тестируйте каждую новую функцию отдельно
  • Используйте систему контроля версий (Git)
  • Создавайте подробные комментарии и документацию
  • Собирайте отзывы от других игроков
  • Балансируйте сложность игры
  • Добавляйте возможность настройки управления
  • Создавайте резервные копии проекта
  • Изучайте код других игр для вдохновения
  • Не бойтесь переписывать код, если это нужно
