Урок 10: Модули и библиотеки

Используем готовый код и расширяем возможности Python

Что такое модули и библиотеки?

Модуль — это файл с кодом Python. Библиотека (пакет) — это набор модулей.

Модуль
Один файл .py
Пакет
Папка с модулями
Библиотека
Набор пакетов
Фреймворк
Готовое решение
# Python поставляется с множеством встроенных модулей
# Их можно использовать без установки

# math - математические функции
import math
print(math.sqrt(25))      # 5.0
print(math.pi)            # 3.141592653589793
print(math.factorial(5))  # 120

# datetime - работа с датами и временем
import datetime
сегодня = datetime.date.today()
print(сегодня)            # 2023-10-15
print(сегодня.year)       # 2023
print(сегодня.month)      # 10

# random - генерация случайных чисел
import random
print(random.randint(1, 10))      # Случайное число от 1 до 10
print(random.choice(['a', 'b', 'c']))  # Случайный выбор

# os - работа с операционной системой
import os
print(os.getcwd())        # Текущая рабочая директория
print(os.listdir('.'))    # Список файлов в директории

Способы импорта модулей

Существует несколько способов импорта модулей:

my_project/
├── main.py
├── utils.py
├── math_operations.py
└── data/
├── __init__.py
└── processor.py 
# 1. Импорт всего модуля
import math
результат = math.sqrt(16)  # Всегда указываем имя модуля

# 2. Импорт с псевдонимом (alias)
import math as m
результат = m.sqrt(16)     # Используем короткое имя

# 3. Импорт конкретных функций
from math import sqrt, pi
результат = sqrt(16)       # Можно использовать без math.
print(pi)                  # Также без math.

# 4. Импорт всех функций (не рекомендуется!)
from math import *
результат = sqrt(16)       # Работает, но может быть конфликт имен

# 5. Импорт из пакетов
# Для пакета data с файлом processor.py
from data.processor import обработать_данные
# или
import data.processor as dp

# 6. Относительный импорт (внутри пакетов)
# В файле data/processor.py:
# from . import другой_модуль  # Точка означает текущий пакет
# from .. import родительский_пакет  # Две точки - родительский пакет

Создание собственных модулей

Любой файл .py можно использовать как модуль:

# Файл geometry.py - наш собственный модуль
"""
Модуль для геометрических вычислений
Автор: Кванториум 22
"""

PI = 3.14159

def площадь_круга(радиус):
    """Вычисляет площадь круга"""
    return PI * радиус ** 2

def площадь_прямоугольника(длина, ширина):
    """Вычисляет площадь прямоугольника"""
    return длина * ширина

def объем_куба(сторона):
    """Вычисляет объем куба"""
    return сторона ** 3

# Файл main.py - основной файл программы
import geometry

# Используем функции из нашего модуля
радиус = 5
площадь = geometry.площадь_круга(радиус)
print(f"Площадь круга радиусом {радиус}: {площадь:.2f}")

# Импорт конкретных функций
from geometry import площадь_прямоугольника, объем_куба

длина = 10
ширина = 5
площадь_прям = площадь_прямоугольника(длина, ширина)
print(f"Площадь прямоугольника {длина}x{ширина}: {площадь_прям}")

# Проверка, является ли файл основным модулем
if __name__ == "__main__":
    # Этот код выполнится только при прямом запуске geometry.py
    print("Тестирование модуля geometry...")
    print(f"Площадь круга радиусом 3: {площадь_круга(3):.2f}")

Создание пакетов

Пакет — это директория с модулями и файлом __init__.py:

# Структура пакета mypackage/
# mypackage/
# ├── __init__.py
# ├── module1.py
# ├── module2.py
# └── subpackage/
#     ├── __init__.py
#     └── module3.py

# Файл mypackage/__init__.py
"""
Пакет mypackage для демонстрации
"""

__version__ = "1.0.0"
__author__ = "Кванториум 22"

# Можно определить, что импортировать при "from mypackage import *"
__all__ = ['module1', 'module2']

# Файл mypackage/module1.py
def приветствие():
    return "Привет из module1!"

class Калькулятор:
    def сложить(self, a, b):
        return a + b

# Файл mypackage/module2.py
def прощание():
    return "Пока из module2!"

# Основная программа
from mypackage import module1
from mypackage.module2 import прощание
import mypackage.subpackage.module3 as mod3

print(module1.приветствие())  # Привет из module1!
print(прощание())            # Пока из module2!

калькулятор = module1.Калькулятор()
print(калькулятор.сложить(5, 3))  # 8

# Импорт из подпакета
результат = mod3.функция_из_подпакета()

Встроенные библиотеки Python

Python включает множество полезных библиотек "из коробки":

Математические

math, decimal, fractions, statistics, random

Дата и время

datetime, time, calendar

Файлы и ОС

os, sys, pathlib, shutil, glob

Разработка

json, csv, re (регулярные выражения), itertools

# Примеры использования встроенных библиотек

# statistics - статистические функции
import statistics
данные = [1, 2, 3, 4, 5, 6, 7, 8, 9]
print(f"Среднее: {statistics.mean(данные)}")
print(f"Медиана: {statistics.median(данные)}")
print(f"Стандартное отклонение: {statistics.stdev(данные):.2f}")

# collections - специальные коллекции
from collections import Counter, defaultdict, deque

# Counter - подсчет элементов
слова = ["яблоко", "банан", "яблоко", "апельсин", "банан", "яблоко"]
счетчик = Counter(слова)
print(f"Яблок: {счетчик['яблоко']}")  # 3
print(f"Все счетчики: {счетчик}")

# defaultdict - словарь со значением по умолчанию
по_классам = defaultdict(list)
по_классам["7А"].append("Анна")
по_классам["7А"].append("Иван")
print(f"7А класс: {по_классам['7А']}")  # ['Анна', 'Иван']
print(f"7Б класс: {по_классам['7Б']}")  # [] (пустой список)

# itertools - итераторы и комбинаторика
from itertools import permutations, combinations, product

буквы = ['A', 'B', 'C']
print("Все перестановки из 2 букв:")
for p in permutations(буквы, 2):
    print(p)

print("\nВсе сочетания из 2 букв:")
for c in combinations(буквы, 2):
    print(c)

# re - регулярные выражения
import re
текст = "Телефоны: +7 999 123-45-67, 8-800-555-35-35"
паттерн = r'\d[\d\s\-\(\)]+'
номера = re.findall(паттерн, текст)
print(f"Найденные номера: {номера}")

Установка внешних библиотек (pip)

pip — менеджер пакетов Python для установки библиотек из интернета.

# Команды в терминале/командной строке:

# Установка библиотеки
# pip install имя_библиотеки

# Установка конкретной версии
# pip install имя_библиотеки==версия

# Установка из файла requirements.txt
# pip install -r requirements.txt

# Просмотр установленных пакетов
# pip list

# Поиск пакетов
# pip search "ключевые слова"

# Обновление пакета
# pip install --upgrade имя_библиотеки

# Удаление пакета
# pip uninstall имя_библиотеки

# Примеры популярных библиотек
# pip install requests      # HTTP запросы
# pip install beautifulsoup4 # Парсинг HTML
# pip install numpy         # Научные вычисления
# pip install pandas        # Анализ данных
# pip install matplotlib    # Графики
# pip install pygame        # Создание игр
# pip install flask         # Веб-фреймворк

# Пример requirements.txt
# requests==2.28.1
# beautifulsoup4==4.11.1
# numpy>=1.21.0
# matplotlib~=3.5.0

Популярные внешние библиотеки

Тысячи библиотек решают самые разные задачи:

# requests - работа с HTTP
import requests

# Отправка GET запроса
ответ = requests.get("https://api.github.com")
print(f"Статус: {ответ.status_code}")  # 200
print(f"Тип контента: {ответ.headers['content-type']}")  # application/json

# Загрузка данных с API
ответ = requests.get("https://jsonplaceholder.typicode.com/posts/1")
данные = ответ.json()
print(f"Заголовок: {данные['title']}")

# beautifulsoup - парсинг HTML
from bs4 import BeautifulSoup

html = """

Пример страницы


Заголовок


Первый абзац


Второй абзац

Ссылка


"""

soup = BeautifulSoup(html, 'html.parser')
print(f"Заголовок: {soup.title.text}")
print(f"Первый абзац: {soup.find('p', class_='text').text}")

# numpy - численные вычисления
import numpy as np

# Создание массивов
массив = np.array([1, 2, 3, 4, 5])
матрица = np.array([[1, 2, 3], [4, 5, 6]])

print(f"Массив: {массив}")
print(f"Матрица:\n{матрица}")
print(f"Сумма: {np.sum(матрица)}")
print(f"Среднее: {np.mean(матрица)}")

# pandas - анализ данных
import pandas as pd

# Создание DataFrame
данные = {
    'Имя': ['Анна', 'Иван', 'Мария'],
    'Возраст': [12, 13, 12],
    'Оценка': [5, 4, 5]
}
df = pd.DataFrame(данные)
print(f"DataFrame:\n{df}")
print(f"\nСредний возраст: {df['Возраст'].mean()}")
print(f"Максимальная оценка: {df['Оценка'].max()}")

Virtual Environments (Виртуальные окружения)

Виртуальные окружения позволяют изолировать зависимости разных проектов. Создаются с помощью venv:

# Создание виртуального окружения
python -m venv venv

# Активация (Windows)
venv\Scripts\activate

# Активация (macOS/Linux)
source venv/bin/activate

# Деактивация
deactivate

Создание и публикация собственных пакетов

Вы можете создать свой пакет и опубликовать его на PyPI:

# Структура проекта для публикации
# mypackage/
# ├── mypackage/
# │   ├── __init__.py
# │   └── main.py
# ├── tests/
# │   └── test_mypackage.py
# ├── setup.py
# ├── README.md
# ├── LICENSE
# └── requirements.txt

# Файл setup.py
from setuptools import setup, find_packages

setup(
    name='mypackage',
    version='1.0.0',
    author='Кванториум 22',
    description='Пример пакета для обучения',
    long_description=open('README.md').read(),
    long_description_content_type='text/markdown',
    url='https://github.com/username/mypackage',
    packages=find_packages(),
    install_requires=[
        'requests>=2.0.0',
    ],
    classifiers=[
        'Programming Language :: Python :: 3',
        'License :: OSI Approved :: MIT License',
        'Operating System :: OS Independent',
    ],
    python_requires='>=3.6',
)

# Команды для публикации
# Установка инструментов
# pip install setuptools wheel twine

# Сборка пакета
# python setup.py sdist bdist_wheel

# Проверка пакета
# twine check dist/*

# Загрузка на Test PyPI (для тестирования)
# twine upload --repository-url https://test.pypi.org/legacy/ dist/*

# Загрузка на PyPI (окончательная)
# twine upload dist/*

# Установка вашего пакета
# pip install mypackage

Практическое задание: "Модульная игра"

Создайте игру с использованием модулей и сторонних библиотек.

Требования:

  1. Создайте структуру проекта:
    • Основной модуль игры
    • Модуль персонажей
    • Модуль предметов и инвентаря
    • Модуль локаций
    • Модуль боевой системы
    • Модуль сохранения игры
  2. Используйте сторонние библиотеки:
    • colorama или termcolor для цветного вывода
    • pygame или curses для графического интерфейса (опционально)
    • json для сохранения игры
    • random для случайных событий
  3. Реализуйте систему плагинов:
    • Возможность добавления новых персонажей через отдельные файлы
    • Загрузка локаций из JSON файлов
    • Динамическая загрузка модулей
  4. Создайте установочный файл setup.py
  5. Напишите документацию для вашего пакета

Пример структуры:

my_game/
├── game/
│   ├── __init__.py
│   ├── main.py          # Основная логика игры
│   ├── character.py     # Классы персонажей
│   ├── inventory.py     # Инвентарь и предметы
│   ├── location.py      # Локации и карта
│   ├── battle.py        # Боевая система
│   └── save.py          # Сохранение/загрузка
├── data/
│   ├── characters.json  # Данные персонажей
│   ├── items.json       # Данные предметов
│   └── locations.json   # Данные локаций
├── plugins/
│   ├── __init__.py
│   └── custom_character.py  # Пример плагина
├── tests/               # Тесты
├── requirements.txt     # Зависимости
├── setup.py            # Файл установки
└── README.md           # Документация

Пример работы программы:

=== ТЕКСТОВАЯ RPG ИГРА ===

Добро пожаловать, герой!
Выберите класс:
1. Воин (сила: 10, ловкость: 5)
2. Маг (сила: 3, ловкость: 7, мана: 10)
3. Лучник (сила: 6, ловкость: 9)

Ваш выбор: 2

Отлично! Вы маг по имени Мерлин.
Вы находитесь в стартовой деревне.

Доступные действия:
1. Осмотреться
2. Открыть инвентарь
3. Переместиться
4. Сохранить игру
5. Выход

Выберите действие: 1

Вы видите:
- Торговца (покупает/продает предметы)
- Таверну (можно отдохнуть)
- Ворота (ведут в темный лес)

Загружен плагин: Новый класс 'Некромант'
Загружена новая локация: 'Подземелье дракона'

Подсказки:

  • Используйте __init__.py для инициализации пакета
  • Для динамической загрузки модулей используйте importlib
  • Создайте базовые классы для персонажей и предметов
  • Используйте декораторы для регистрации плагинов
  • Добавьте обработку ошибок при загрузке модулей
  • Используйте logging для отладки
  • Создайте виртуальное окружение для проекта

Полезные советы

  • Давайте модулям понятные имена в snake_case
  • Всегда добавляйте документацию (docstrings)
  • Используйте виртуальные окружения для каждого проекта
  • Храните зависимости в requirements.txt
  • Разделяйте код на логические модули
  • Избегайте циклических импортов
  • Используйте if __name__ == "__main__" для тестового кода
  • Пишите тесты для ваших модулей
  • Проверяйте совместимость версий библиотек
