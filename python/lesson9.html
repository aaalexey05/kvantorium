Урок 9: Работа с файлами

Храним данные между запусками программы

Зачем нужны файлы?

Файлы позволяют сохранять данные после завершения программы и читать данные из внешних источников.

Текстовые файлы
.txt, .csv, .json
Бинарные файлы
.jpg, .png, .exe
Структурированные
.json, .xml, .yaml
Чтение: Загрузка данных из файла в программу
Запись: Сохранение данных из программы в файл
Добавление: Добавление новых данных в конец файла

Открытие и закрытие файлов

Работа с файлами всегда состоит из трех этапов: открыть → работать → закрыть.

# Способ 1: Явное открытие и закрытие (устаревший способ)
файл = open("пример.txt", "r", encoding="utf-8")
содержимое = файл.read()
файл.close()  # Важно не забыть закрыть!

# Способ 2: Использование with (рекомендуемый способ)
with open("пример.txt", "r", encoding="utf-8") as файл:
    содержимое = файл.read()
# Файл автоматически закроется при выходе из блока with

# Параметры функции open()
# 'r' - чтение (по умолчанию)
# 'w' - запись (стирает существующий файл)
# 'a' - добавление в конец
# 'x' - создание нового файла (ошибка если файл существует)
# 'b' - бинарный режим
# 't' - текстовый режим (по умолчанию)
# '+' - чтение и запись

# Примеры режимов открытия
with open("данные.txt", "w", encoding="utf-8") as файл:  # Запись
    файл.write("Новые данные")

with open("данные.txt", "a", encoding="utf-8") as файл:  # Добавление
    файл.write("\nЕще данные")

with open("данные.txt", "r+", encoding="utf-8") as файл:  # Чтение и запись
    содержимое = файл.read()
    файл.write("\nДополнительные данные")

Чтение файлов

Существует несколько способов чтения данных из файла:

# Создадим тестовый файл
with open("тест.txt", "w", encoding="utf-8") as файл:
    файл.write("Первая строка\n")
    файл.write("Вторая строка\n")
    файл.write("Третья строка\n")

# 1. read() - чтение всего файла
with open("тест.txt", "r", encoding="utf-8") as файл:
    весь_текст = файл.read()
    print("Весь текст:")
    print(весь_текст)

# 2. readline() - чтение по одной строке
with open("тест.txt", "r", encoding="utf-8") as файл:
    print("\nПострочное чтение:")
    строка1 = файл.readline()
    строка2 = файл.readline()
    print(f"Строка 1: {строка1.strip()}")
    print(f"Строка 2: {строка2.strip()}")

# 3. readlines() - чтение всех строк в список
with open("тест.txt", "r", encoding="utf-8") as файл:
    все_строки = файл.readlines()
    print("\nВсе строки как список:")
    for строка in все_строки:
        print(f"→ {строка.strip()}")

# 4. Итерация по файлу (самый эффективный способ)
with open("тест.txt", "r", encoding="utf-8") as файл:
    print("\nИтерация по файлу:")
    for номер, строка in enumerate(файл, 1):
        print(f"Строка {номер}: {строка.strip()}")

# 5. Чтение определенного количества символов
with open("тест.txt", "r", encoding="utf-8") as файл:
    первые_10 = файл.read(10)
    print(f"\nПервые 10 символов: '{первые_10}'")

# 6. seek() - перемещение по файлу
with open("тест.txt", "r", encoding="utf-8") as файл:
    файл.seek(15)  # Перемещаемся на 15-й символ
    оставшийся_текст = файл.read()
    print(f"\nТекст с 15-го символа: {оставшийся_текст[:30]}...")

Запись в файлы

Запись данных в файл может быть разной: перезапись, добавление, форматирование.

# 1. Запись строки
with open("запись.txt", "w", encoding="utf-8") as файл:
    файл.write("Привет, мир!\n")
    файл.write("Это тест записи в файл.\n")

# 2. Добавление в конец
with open("запись.txt", "a", encoding="utf-8") as файл:
    файл.write("Эта строка добавлена позже.\n")

# 3. Запись нескольких строк
строки = ["Строка 1", "Строка 2", "Строка 3"]
with open("много_строк.txt", "w", encoding="utf-8") as файл:
    for строка in строки:
        файл.write(строка + "\n")

# 4. writelines() - запись списка строк
новые_строки = ["Еще одна\n", "И еще\n", "Последняя\n"]
with open("список_строк.txt", "w", encoding="utf-8") as файл:
    файл.writelines(новые_строки)

# 5. Форматированный вывод
данные = [
    {"имя": "Анна", "возраст": 12, "оценка": 5},
    {"имя": "Иван", "возраст": 13, "оценка": 4},
    {"имя": "Мария", "возраст": 12, "оценка": 5}
]

with open("ученики.csv", "w", encoding="utf-8") as файл:
    # Заголовок
    файл.write("Имя;Возраст;Оценка\n")
    
    # Данные
    for ученик in данные:
        строка = f"{ученик['имя']};{ученик['возраст']};{ученик['оценка']}\n"
        файл.write(строка)

# 6. Проверка записи
with open("ученики.csv", "r", encoding="utf-8") as файл:
    print("Содержимое файла ученики.csv:")
    print(файл.read())

Работа с CSV файлами

CSV (Comma-Separated Values) - популярный формат для табличных данных.

import csv

# Данные для записи
ученики = [
    ["Имя", "Возраст", "Класс", "Средний балл"],
    ["Анна", "12", "7А", "4.8"],
    ["Иван", "13", "7Б", "4.2"],
    ["Мария", "12", "7А", "4.9"],
    ["Сергей", "13", "7Б", "3.9"]
]

# Запись в CSV
with open("ученики.csv", "w", encoding="utf-8", newline='') as файл:
    писатель = csv.writer(файл, delimiter=';')
    писатель.writerows(ученики)

print("Файл ученики.csv создан")

# Чтение из CSV
with open("ученики.csv", "r", encoding="utf-8") as файл:
    читатель = csv.reader(файл, delimiter=';')
    print("\nСодержимое CSV файла:")
    for строка in читатель:
        print(строка)

# Работа со словарями через DictReader/DictWriter
новые_ученики = [
    {"имя": "Ольга", "возраст": "12", "класс": "7А", "средний_балл": "4.5"},
    {"имя": "Дмитрий", "возраст": "13", "класс": "7Б", "средний_балл": "4.0"}
]

# Запись словарей
with open("ученики_dict.csv", "w", encoding="utf-8", newline='') as файл:
    поля = ["имя", "возраст", "класс", "средний_балл"]
    писатель = csv.DictWriter(файл, fieldnames=поля, delimiter=',')
    писатель.writeheader()
    писатель.writerows(новые_ученики)

# Чтение словарей
with open("ученики_dict.csv", "r", encoding="utf-8") as файл:
    читатель = csv.DictReader(файл)
    print("\nЧтение как словари:")
    for строка in читатель:
        print(f"{строка['имя']} - {строка['класс']} класс, балл: {строка['средний_балл']}")

Работа с JSON файлами

JSON (JavaScript Object Notation) - популярный формат для обмена структурированными данными.

import json

# Данные для записи в JSON
школа = {
    "название": "Школа №123",
    "адрес": "ул. Пушкина, 10",
    "ученики": [
        {"имя": "Анна", "возраст": 12, "класс": "7А", "оценки": [5, 4, 5, 4, 5]},
        {"имя": "Иван", "возраст": 13, "класс": "7Б", "оценки": [4, 4, 3, 5, 4]},
        {"имя": "Мария", "возраст": 12, "класс": "7А", "оценки": [5, 5, 5, 5, 5]}
    ],
    "статистика": {
        "всего_учеников": 3,
        "средний_возраст": 12.33,
        "средний_балл": 4.53
    }
}

# Запись в JSON файл
with open("школа.json", "w", encoding="utf-8") as файл:
    json.dump(школа, файл, ensure_ascii=False, indent=2)

print("Файл школа.json создан")

# Чтение из JSON файла
with open("школа.json", "r", encoding="utf-8") as файл:
    данные = json.load(файл)

print("\nДанные из JSON файла:")
print(f"Название школы: {данные['название']}")
print(f"Всего учеников: {данные['статистика']['всего_учеников']}")

print("\nУченики:")
for ученик in данные['ученики']:
    средний = sum(ученик['оценки']) / len(ученик['оценки'])
    print(f"- {ученик['имя']} ({ученик['класс']}): средний балл {средний:.2f}")

# Преобразование строки JSON в объект Python
json_строка = '{"имя": "Анна", "возраст": 12, "хобби": ["чтение", "программирование"]}'
объект = json.loads(json_строка)
print(f"\nИз строки JSON: {объект['имя']}, хобби: {', '.join(объект['хобби'])}")

# Преобразование объекта Python в строку JSON
новый_объект = {"тест": "данные", "число": 42, "список": [1, 2, 3]}
json_строка_новая = json.dumps(новый_объект, ensure_ascii=False, indent=2)
print(f"\nВ строку JSON:\n{json_строка_новая}")

Работа с путями файлов и каталогами

Библиотека os предоставляет функции для работы с файловой системой.

import os
import shutil

# Получение информации о файлах и каталогах
print("Текущая рабочая директория:", os.getcwd())

# Проверка существования
if os.path.exists("школа.json"):
    print("Файл школа.json существует")
    
    # Получение информации о файле
    размер = os.path.getsize("школа.json")
    время_изменения = os.path.getmtime("школа.json")
    print(f"Размер файла: {размер} байт")
    print(f"Время изменения: {время_изменения}")

# Список файлов в директории
print("\nФайлы в текущей директории:")
for файл in os.listdir("."):
    if os.path.isfile(файл):
        print(f"  Файл: {файл}")
    elif os.path.isdir(файл):
        print(f"  Папка: {файл}")

# Создание и удаление директорий
if not os.path.exists("мои_файлы"):
    os.mkdir("мои_файлы")
    print("Создана папка 'мои_файлы'")

# Копирование файлов
shutil.copy("школа.json", "мои_файлы/копия_школа.json")
print("Файл скопирован в мои_файлы")

# Переименование
if os.path.exists("запись.txt"):
    os.rename("запись.txt", "старая_запись.txt")
    print("Файл переименован")

# Удаление файлов
if os.path.exists("старая_запись.txt"):
    os.remove("старая_запись.txt")
    print("Файл удален")

# Рекурсивный обход директорий
print("\nРекурсивный обход:")
for корень, директории, файлы in os.walk("."):
    уровень = корень.count(os.sep) - ".".count(os.sep)
    отступ = " " * 4 * уровень
    print(f"{отступ}[{os.path.basename(корень)}]")
    подотступ = " " * 4 * (уровень + 1)
    for файл in файлы:
        print(f"{подотступ}{файл}")

Бинарные файлы

Работа с изображениями, аудио и другими нетекстовыми файлами.

# Копирование бинарного файла (например, изображения)
try:
    with open("исходное_изображение.jpg", "rb") as исходный:
        with open("копия_изображения.jpg", "wb") as копия:
            # Читаем и записываем блоками для экономии памяти
            while True:
                блок = исходный.read(4096)  # 4KB блок
                if not блок:
                    break
                копия.write(блок)
    print("Бинарный файл скопирован")
except FileNotFoundError:
    print("Исходный файл не найден, создадим тестовый бинарный файл")
    
    # Создадим простой бинарный файл
    with open("тест.bin", "wb") as файл:
        # Запишем байты
        данные = bytes([65, 66, 67, 68, 69])  # ABCDE в ASCII
        файл.write(данные)
        файл.write(b"\nЭто текст в бинарном файле\n")

# Чтение бинарного файла
with open("тест.bin", "rb") as файл:
    # Читаем первые 5 байт
    первые_5_байт = файл.read(5)
    print(f"Первые 5 байт: {первые_5_байт}")
    print(f"Как символы: {первые_5_байт.decode('ascii')}")
    
    # Позиционируемся и читаем дальше
    файл.seek(0)  # Возвращаемся в начало
    весь_файл = файл.read()
    print(f"\nВесь файл как байты: {весь_файл}")
    print(f"Весь файл как текст: {весь_файл.decode('utf-8')}")

# Работа с различными кодировками
текст = "Привет, мир! 🌍"

# Запись в разных кодировках
with open("utf8.txt", "w", encoding="utf-8") as файл:
    файл.write(текст)

with open("cp1251.txt", "w", encoding="cp1251") as файл:
    файл.write(текст)

# Чтение с указанием кодировки
with open("utf8.txt", "r", encoding="utf-8") as файл:
    print(f"\nUTF-8: {файл.read()}")

with open("cp1251.txt", "r", encoding="cp1251") as файл:
    print(f"CP1251: {файл.read()}")

Практическое задание: "Персональный дневник"

Создайте программу для ведения персонального дневника с сохранением в файлы.

Требования:

  1. Реализуйте функции:
    • Добавить запись в дневник
    • Просмотреть все записи
    • Поиск по дате или ключевым словам
    • Редактирование записей
    • Удаление записей
    • Статистика (сколько записей, по дням и т.д.)
  2. Храните данные в JSON формате:
    • Каждая запись должна иметь дату, заголовок, текст и теги
    • Поддерживайте историю изменений
  3. Добавьте возможность экспорта:
    • В TXT (простые текстовые файлы)
    • В CSV (для анализа в табличных редакторах)
    • В HTML (красивое отображение)
  4. Реализуйте резервное копирование:
    • Автоматическое копирование при изменении
    • Восстановление из резервной копии
    • Хранение нескольких версий
  5. Добавьте работу с каталогами:
    • Создание папки для каждого месяца
    • Сортировка по годам
    • Архивация старых записей

Подсказки:

  • Используйте модуль datetime для работы с датами
  • Храните каждую запись как словарь в списке
  • Используйте UUID для уникальных идентификаторов записей
  • Реализуйте автодополнение тегов
  • Добавьте возможность прикрепления файлов к записям

  • Добавьте шифрование для приватных записей

Полезные советы

  • Всегда используйте with open() as для работы с файлами
  • Указывайте кодировку (encoding="utf-8") при работе с текстовыми файлами
  • Используйте newline='' при работе с CSV файлами в Windows
  • Для больших файлов читайте данные блоками, а не весь файл сразу
  • Регулярно делайте резервные копии важных данных
  • Проверяйте существование файлов перед операциями с ними
  • Используйте JSON для сложных структур данных
  • Для бинарных файлов всегда используйте режимы 'rb' и 'wb'
