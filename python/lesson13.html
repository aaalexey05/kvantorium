Урок 13: Библиотека Turtle

Создаем графику и анимации с помощью черепашки

Что такое Turtle Graphics?

Turtle — это библиотека Python для создания графики и анимаций. Представьте себе черепашку с пером, которая рисует линии, когда двигается.

Визуализация Turtle

Основные команды

  • forward(distance) - движение вперед
  • backward(distance) - движение назад
  • left(angle) - поворот влево
  • right(angle) - поворот вправо
  • penup() - поднять перо
  • pendown() - опустить перо
  • color("red") - изменить цвет
  • pensize(5) - толщина линии
  • circle(50) - нарисовать круг
# Простейшая программа на Turtle
import turtle

# Создаем черепашку
t = turtle.Turtle()

# Настраиваем черепашку
t.shape("turtle")  # Делаем курсор в виде черепашки
t.color("green")
t.pensize(3)

# Рисуем квадрат
for i in range(4):
    t.forward(100)  # Двигаемся вперед на 100 пикселей
    t.left(90)      # Поворачиваем на 90 градусов влево

# Перемещаемся
t.penup()          # Поднимаем перо (перестаем рисовать)
t.goto(150, 0)     # Перемещаемся в другую точку
t.pendown()        # Опускаем перо (начинаем рисовать)

# Рисуем круг
t.color("red")
t.circle(50)       # Рисуем круг радиусом 50

# Завершаем программу
turtle.done()

import turtle

screen = turtle.Screen()
screen.title("Моя графика на Turtle")  # Заголовок окна
screen.bgcolor("lightblue")            # Цвет фона

# Настройка черепашки
t = turtle.Turtle()
t.speed(5)           # Скорость от 1 (медленно) до 10 (быстро), 0 - максимальная

# Изменяем форму курсора
# Доступные формы: "arrow", "turtle", "circle", "square", "triangle", "classic"
t.shape("turtle")

# Изменяем цвет черепашки и пера
t.color("darkgreen", "lightgreen")  # Первый цвет - линии, второй - заливка

# Изменяем размер пера
t.pensize(3)

# Поднимаем перо (чтобы не рисовать при перемещении)
t.penup()

# Перемещаемся в начальную позицию
t.goto(-200, 0)

# Опускаем перо (начинаем рисовать)
t.pendown()

# Рисуем линию
t.forward(100)

# Скрываем/показываем черепашку

# Получаем текущее направление

Рисование фигур

Создадим различные геометрические фигуры:

import turtle

t = turtle.Turtle()
t.speed(5)
t.pensize(2)

# 1. Квадрат
def draw_square(size, color="black"):
    t.color(color)
    for _ in range(4):
        t.forward(size)
        t.left(90)

# 2. Треугольник
def draw_triangle(size, color="black"):
    t.color(color)
    for _ in range(3):
        t.forward(size)
        t.left(120)

# 3. Круг
def draw_circle(radius, color="black"):
    t.color(color)
    t.circle(radius)

# 4. Многоугольник
def draw_polygon(sides, size, color="black"):
    t.color(color)
    angle = 360 / sides
    for _ in range(sides):
        t.forward(size)
        t.left(angle)

# 5. Звезда
def draw_star(size, color="black"):
    t.color(color)
    for _ in range(5):
        t.forward(size)
        t.right(144)

# Рисуем фигуры в разных позициях
# Перемещаемся и рисуем квадрат
t.penup()
t.goto(-300, 100)
t.pendown()
draw_square(100, "red")

# Перемещаемся и рисуем треугольник
t.penup()
t.goto(-150, 100)
t.pendown()
draw_triangle(100, "blue")

# Перемещаемся и рисуем круг
t.penup()
t.goto(0, 100)
t.pendown()
draw_circle(50, "green")

# Перемещаемся и рисуем шестиугольник
t.penup()
t.goto(150, 100)
t.pendown()
draw_polygon(6, 60, "purple")

# Перемещаемся и рисуем звезду
t.penup()
t.goto(-300, -100)
t.pendown()
draw_star(100, "orange")

# Перемещаемся и рисуем спираль
def draw_spiral():
    t.penup()
    t.goto(150, -100)
    t.pendown()
    
    t.color("brown")
    length = 5
    for i in range(50):
        t.forward(length)
        t.right(91)  # Не 90 градусов для создания спирали
        length += 3

draw_spiral()

turtle.done()

def draw_filled_circle(radius, color):
    t.penup()
    t.goto(0, 200)
    t.pendown()
    
    t.color(color)
    t.begin_fill()      # Начинаем заливку
    t.circle(radius)
    t.end_fill()        # Заканчиваем заливку

draw_filled_circle(50, "lightblue")

# Градиентный квадрат
def draw_gradient_square(size):
    t.penup()
    t.goto(-200, -100)
    t.pendown()
    
    # Меняем цвет постепенно
    for i in range(4):
        # Вычисляем цвет (от черного к красному)
        red_value = i * 0.25
        t.color(red_value, 0, 0)
        
        # Рисуем сторону
        t.forward(size)
        t.left(90)
        
        # Увеличиваем толщину пера
        t.pensize(i + 2)

draw_gradient_square(80)

# Разноцветная спираль
def draw_colorful_spiral():
    t.penup()
    t.goto(100, -100)
    t.pendown()
    
    t.pensize(2)
    colors = ["red", "orange", "yellow", "green", "blue", "purple"]
    
    for i in range(100):
        # Меняем цвет каждые 10 шагов
        t.color(colors[i // 10 % len(colors)])
        t.forward(i * 2)
        t.right(91)

draw_colorful_spiral()

            t.left(120)
        t.end_fill()

        t.left(10)

draw_flower()

Анимации и движение

Создадим анимации и интерактивные элементы:

import turtle
import time
import random

screen = turtle.Screen()
screen.title("Анимации на Turtle")
screen.bgcolor("black")
screen.tracer(0)  # Отключаем автообновление для плавной анимации

# 1. Движущаяся черепашка
moving_turtle = turtle.Turtle()
moving_turtle.shape("turtle")
moving_turtle.color("cyan")
moving_turtle.penup()
moving_turtle.goto(-300, 0)

# 2. Мигающая звезда
blinking_star = turtle.Turtle()
blinking_star.shape("circle")
blinking_star.color("yellow")
blinking_star.penup()
blinking_star.goto(200, 200)

# 3. Вращающийся квадрат
rotating_square = turtle.Turtle()
rotating_square.color("magenta")
rotating_square.penup()
rotating_square.goto(-200, -200)
rotating_square.pendown()

# 4. Падающий снег
snowflakes = []
for _ in range(50):
    snowflake = turtle.Turtle()
    snowflake.shape("circle")
    snowflake.color("white")
    snowflake.shapesize(0.5)
    snowflake.penup()
    snowflake.goto(random.randint(-400, 400), random.randint(300, 400))
    snowflake.speed = random.uniform(1, 3)
    snowflakes.append(snowflake)

# Анимационный цикл
def animate():
    frame = 0
    
    while True:
        screen.update()  # Обновляем экран
        time.sleep(0.03)  # Небольшая задержка для плавности
        frame += 1
        
        # 1. Движение черепашки
        moving_turtle.forward(3)
        if moving_turtle.xcor() > 400:
            moving_turtle.goto(-400, random.randint(-200, 200))
        
        # 2. Мигание звезды
        if frame % 10 == 0:
            if blinking_star.isvisible():
                blinking_star.hideturtle()
            else:
                blinking_star.showturtle()
        
        # 3. Вращение квадрата
        rotating_square.clear()
        rotating_square.setheading(frame * 2)  # Поворачиваем квадрат
        
        # Рисуем квадрат в новом положении
        for _ in range(4):
            rotating_square.forward(50)
            rotating_square.right(90)
        
        # 4. Падающий снег
        for snowflake in snowflakes:
            snowflake.sety(snowflake.ycor() - snowflake.speed)
            
            # Если снежинка упала, возвращаем ее наверх
            if snowflake.ycor() < -300:
                snowflake.goto(random.randint(-400, 400), 300)
                snowflake.speed = random.uniform(1, 3)
        
        # Добавляем случайные вспышки
        if frame % 20 == 0:
            flash = turtle.Turtle()
            flash.shape("circle")
            flash.color(random.choice(["red", "green", "blue", "yellow"]))
            flash.penup()
            flash.goto(random.randint(-350, 350), random.randint(-250, 250))
            flash.shapesize(random.uniform(0.5, 2))
            flash.speed = 0
            
            # Создаем анимацию исчезновения
            def fade_out(t):
                for _ in range(10):
                    t.shapesize(t.shapesize()[0] * 0.7)
                    screen.update()
                    time.sleep(0.05)
                t.hideturtle()
                t.clear()
            
            # Запускаем анимацию в отдельном потоке
            import threading
            threading.Thread(target=fade_out, args=(flash,)).start()

# Создаем интерактивную черепашку
interactive_turtle = turtle.Turtle()
interactive_turtle.shape("turtle")
interactive_turtle.color("lime")
interactive_turtle.penup()
interactive_turtle.goto(0, 0)

# Функции для управления
def move_up():
    interactive_turtle.sety(interactive_turtle.ycor() + 20)

def move_down():
    interactive_turtle.sety(interactive_turtle.ycor() - 20)

def move_left():
    interactive_turtle.setx(interactive_turtle.xcor() - 20)

def move_right():
    interactive_turtle.setx(interactive_turtle.xcor() + 20)

def change_color():
    colors = ["red", "green", "blue", "yellow", "purple", "orange"]
    interactive_turtle.color(random.choice(colors))

def draw_circle():
    interactive_turtle.pendown()
    interactive_turtle.circle(30)
    interactive_turtle.penup()

# Назначаем клавиши управления
screen.listen()
screen.onkeypress(move_up, "Up")
screen.onkeypress(move_down, "Down")
screen.onkeypress(move_left, "Left")
screen.onkeypress(move_right, "Right")
screen.onkeypress(change_color, "c")
screen.onkeypress(draw_circle, "space")

# Запускаем анимацию в отдельном потоке
import threading
animation_thread = threading.Thread(target=animate)
animation_thread.daemon = True
animation_thread.start()

turtle.done()

Создание игр на Turtle

Разработаем простые игры с использованием Turtle:

import turtle
import random
import time

# ========== ИГРА 1: "Лабиринт" ==========
def maze_game():
    screen = turtle.Screen()
    screen.title("Лабиринт")
    screen.bgcolor("darkblue")
    screen.setup(600, 600)
    
    # Игрок
    player = turtle.Turtle()
    player.shape("turtle")
    player.color("green")
    player.penup()
    player.goto(-250, -250)
    
    # Финиш
    finish = turtle.Turtle()
    finish.shape("circle")
    finish.color("gold")
    finish.penup()
    finish.goto(250, 250)
    finish.shapesize(2)
    
    # Стены лабиринта
    walls = []
    
    # Вертикальные стены
    for y in range(-200, 300, 100):
        wall = turtle.Turtle()
        wall.shape("square")
        wall.color("gray")
        wall.penup()
        wall.shapesize(stretch_wid=5, stretch_len=0.5)
        wall.goto(random.randint(-150, 150), y)
        walls.append(wall)
    
    # Горизонтальные стены
    for x in range(-200, 300, 100):
        wall = turtle.Turtle()
        wall.shape("square")
        wall.color("gray")
        wall.penup()
        wall.shapesize(stretch_wid=0.5, stretch_len=5)
        wall.goto(x, random.randint(-150, 150))
        walls.append(wall)
    
    # Управление
    def move_up():
        player.sety(player.ycor() + 10)
        check_collision()
        check_win()
    
    def move_down():
        player.sety(player.ycor() - 10)
        check_collision()
        check_win()
    
    def move_left():
        player.setx(player.xcor() - 10)
        check_collision()
        check_win()
    
    def move_right():
        player.setx(player.xcor() + 10)
        check_collision()
        check_win()
    
    # Проверка столкновений
    def check_collision():
        for wall in walls:
            if (abs(player.xcor() - wall.xcor()) < 30 and 
                abs(player.ycor() - wall.ycor()) < 30):
                # Столкновение - возвращаем игрока назад
                player.goto(-250, -250)
                return
    
    # Проверка победы
    def check_win():
        if (abs(player.xcor() - finish.xcor()) < 20 and 
            abs(player.ycor() - finish.ycor()) < 20):
            finish.color("red")
            player.hideturtle()
            # Поздравление
            congrats = turtle.Turtle()
            congrats.hideturtle()
            congrats.penup()
            congrats.goto(0, 0)
            congrats.color("white")
            congrats.write("ПОБЕДА!", align="center", font=("Arial", 32, "bold"))
    
    # Назначение клавиш
    screen.listen()
    screen.onkeypress(move_up, "Up")
    screen.onkeypress(move_down, "Down")
    screen.onkeypress(move_left, "Left")
    screen.onkeypress(move_right, "Right")
    
    turtle.done()

def snake_game():
    screen = turtle.Screen()
    screen.title("Змейка")
    screen.bgcolor("black")
    screen.setup(600, 600)
    screen.tracer(0)
    
    # Голова змейки
    head = turtle.Turtle()
    head.shape("square")
    head.color("green")
    head.penup()
    head.goto(0, 0)
    head.direction = "stop"
    
    # Еда
    food = turtle.Turtle()
    food.shape("circle")
    food.color("red")
    food.penup()
    food.goto(0, 100)
    
    # Сегменты змейки
    segments = []
    
    # Счет
    score = 0
    score_display = turtle.Turtle()
    score_display.speed(0)
    score_display.color("white")
    score_display.penup()
    score_display.hideturtle()
    score_display.goto(0, 260)
    score_display.write(f"Счет: {score}", align="center", font=("Arial", 24, "normal"))
    
    # Функции управления
    def go_up():
        if head.direction != "down":
            head.direction = "up"
    
    def go_down():
        if head.direction != "up":
            head.direction = "down"
    
    def go_left():
        if head.direction != "right":
            head.direction = "left"
    
    def go_right():
        if head.direction != "left":
            head.direction = "right"
    
    def move():
        if head.direction == "up":
            y = head.ycor()
            head.sety(y + 20)
        
        if head.direction == "down":
            y = head.ycor()
            head.sety(y - 20)
        
        if head.direction == "left":
            x = head.xcor()
            head.setx(x - 20)
        
        if head.direction == "right":
            x = head.xcor()
            head.setx(x + 20)
    
    # Проверка столкновения с едой
    def check_food_collision():
        if head.distance(food) < 20:
            # Перемещаем еду в случайное место
            x = random.randint(-280, 280)
            y = random.randint(-280, 280)
            food.goto(x, y)
            
            # Добавляем новый сегмент
            new_segment = turtle.Turtle()
            new_segment.shape("square")
            new_segment.color("lightgreen")
            new_segment.penup()
            segments.append(new_segment)
            
            # Увеличиваем счет
            global score
            score += 10
            score_display.clear()
            score_display.write(f"Счет: {score}", align="center", font=("Arial", 24, "normal"))
    
    # Движение сегментов
    def move_segments():
        # Двигаем сегменты от конца к началу
        for i in range(len(segments) - 1, 0, -1):
            x = segments[i-1].xcor()
            y = segments[i-1].ycor()
            segments[i].goto(x, y)
        
        # Двигаем первый сегмент к голове
        if len(segments) > 0:
            x = head.xcor()
            y = head.ycor()
            segments[0].goto(x, y)
    
    # Проверка столкновений
    def check_collisions():
        # Столкновение с границами
        if (head.xcor() > 290 or head.xcor() < -290 or 
            head.ycor() > 290 or head.ycor() < -290):
            game_over()
        
        # Столкновение с собственным телом
        for segment in segments:
            if segment.distance(head) < 20:
                game_over()
    
    def game_over():
        time.sleep(1)
        head.goto(0, 0)
        head.direction = "stop"
        
        # Скрываем сегменты
        for segment in segments:
            segment.goto(1000, 1000)
        
        segments.clear()
        
        # Сбрасываем счет
        global score
        score = 0
        score_display.clear()
        score_display.write(f"Счет: {score}", align="center", font=("Arial", 24, "normal"))
        
        # Сообщение о проигрыше
        game_over_display = turtle.Turtle()
        game_over_display.color("red")
        game_over_display.penup()
        game_over_display.hideturtle()
        game_over_display.goto(0, 0)
        game_over_display.write("ИГРА ОКОНЧЕНА", align="center", font=("Arial", 36, "bold"))
        time.sleep(2)
        game_over_display.clear()
    
    # Назначение клавиш
    screen.listen()
    screen.onkeypress(go_up, "Up")
    screen.onkeypress(go_down, "Down")
    screen.onkeypress(go_left, "Left")
    screen.onkeypress(go_right, "Right")
    
    # Главный игровой цикл
    while True:
        screen.update()
        move()
        check_food_collision()
        move_segments()
        check_collisions()
        time.sleep(0.1)

    score_display.goto(0, 260)
    score_display.write(f"Счет: {score}", align="center", font=("Arial", 24, "normal"))
    
    # Движение платформы
    def paddle_right():
        x = paddle.xcor()
        if x < 350:
            paddle.setx(x + 20)
    
    def paddle_left():
        x = paddle.xcor()
        if x > -350:
            paddle.setx(x - 20)
    
    # Главный игровой цикл
    def game_loop():
        nonlocal score
        
        # Двигаем мяч
        ball.setx(ball.xcor() + ball.dx)
        ball.sety(ball.ycor() + ball.dy)
        
        # Отскок от стен
        if ball.xcor() > 390:
            ball.setx(390)
            ball.dx *= -1
        
        if ball.xcor() < -390:
            ball.setx(-390)
            ball.dx *= -1
        
        if ball.ycor() > 290:
            ball.sety(290)
            ball.dy *= -1
        
        # Отскок от платформы
        if (ball.ycor() < -240 and ball.ycor() > -250 and 
            ball.xcor() < paddle.xcor() + 50 and ball.xcor() > paddle.xcor() - 50):
            ball.sety(-240)
            ball.dy *= -1
        
        # Проверка столкновения с кирпичами
        for brick in bricks:
            if (ball.distance(brick) < 30 and brick.isvisible()):
                brick.hideturtle()
                ball.dy *= -1
                score += 10
                score_display.clear()
                score_display.write(f"Счет: {score}", align="center", font=("Arial", 24, "normal"))
                
                # Проверка победы
                if all(not brick.isvisible() for brick in bricks):
                    win_display = turtle.Turtle()
                    win_display.color("gold")
                    win_display.penup()
                    win_display.hideturtle()
                    win_display.goto(0, 0)
                    win_display.write("ПОБЕДА!", align="center", font=("Arial", 48, "bold"))
                    return
        
        # Проверка проигрыша
        if ball.ycor() < -290:
            game_over_display = turtle.Turtle()
            game_over_display.color("red")
            game_over_display.penup()
            game_over_display.hideturtle()
            game_over_display.goto(0, 0)
            game_over_display.write("ИГРА ОКОНЧЕНА", align="center", font=("Arial", 48, "bold"))
            return
        
        screen.update()
        screen.ontimer(game_loop, 10)
    
    # Назначение клавиш
    screen.listen()
    screen.onkeypress(paddle_right, "Right")
    screen.onkeypress(paddle_left, "Left")
    
    # Запуск игры
    game_loop()
    turtle.done()

# Меню выбора игры
